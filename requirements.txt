import streamlit as st
import pandas as pd
import re
from datetime import datetime

# =====================================================================
# 1. CONFIGURAÇÕES E CSS (MOTOR VISUAL FLEXBOX)
# =====================================================================
st.set_page_config(page_title="🛡️ Gestão de Calibração", layout="wide")

st.markdown("""
    <style>
    .card { background-color:#1E2130; padding:20px; border-radius:12px; border-top:5px solid #4a4f63; margin-bottom:10px; transition: 0.3s; }
    .card:hover { transform: scale(1.02); box-shadow: 0 10px 20px rgba(0,0,0,0.3); }
    .status-badge { padding:5px 10px; border-radius:5px; font-weight:bold; color:white; display:inline-block; }
    </style>
""", unsafe_allow_html=True)

# =====================================================================
# 2. LÓGICA DE PROCESSAMENTO (PARSER DE DADOS)
# =====================================================================
@st.cache_data
def carregar_e_processar_dados():
    # Carrega seu CSV (futuramente trocar por st.connection('gsheets'))
    df = pd.read_csv("compras,_estoque_e_producao_274482357544975.csv", sep=';')
    
    # Extração de Data via Regex (Procura formato dd/mm/aaaa no campo Características)
    def extrair_data(texto):
        if pd.isna(texto): return None
        match = re.search(r'\d{2}/\d{2}/\d{4}', str(texto))
        return pd.to_datetime(match.group(0), dayfirst=True) if match else None

    df['DATA_PROXIMA'] = df['Características'].apply(extrair_data)
    
    # Cálculo de Status em relação à data atual (22/04/2026)
    hoje = datetime(2026, 4, 22)
    df['DIAS_RESTANTES'] = (df['DATA_PROXIMA'] - hoje).dt.days
    
    def rotular_status(dias):
        if pd.isna(dias): return "SEM DATA"
        if dias < 0: return "VENCIDO"
        if dias <= 30: return "PRÓXIMO VENCIMENTO"
        return "NO PRAZO"
    
    df['STATUS'] = df['DIAS_RESTANTES'].apply(rotular_status)
    return df

# =====================================================================
# 3. INTERFACE E DASHBOARD
# =====================================================================
df = carregar_e_processar_dados()

st.title("⚖️ GESTÃO DE INSTRUMENTOS")

# Filtros Sidebar
st.sidebar.header("🔍 FILTROS")
status_filter = st.sidebar.multiselect("Status:", df['STATUS'].unique(), default=df['STATUS'].unique())
df_filt = df[df['STATUS'].isin(status_filter)]

# KPIs
c1, c2, c3 = st.columns(3)
c1.metric("Total de Itens", len(df_filt))
c2.metric("Vencidos", len(df_filt[df_filt['STATUS'] == 'VENCIDO']))
c3.metric("Próximos Vencimentos", len(df_filt[df_filt['STATUS'] == 'PRÓXIMO VENCIMENTO']))

# Exibição (Cards Blindados)
for _, row in df_filt.iterrows():
    cor = "#ff4b4b" if row['STATUS'] == "VENCIDO" else ("#F1C40F" if row['STATUS'] == "PRÓXIMO VENCIMENTO" else "#2ecc71")
    
    with st.container():
        st.markdown(f"""
        <div class='card' style='border-top-color:{cor};'>
            <div style='display:flex; justify-content:space-between;'>
                <h3>{row['Descrição']}</h3>
                <span class='status-badge' style='background-color:{cor};'>{row['STATUS']}</span>
            </div>
            <p><b>Tag/Código:</b> {row['Código']} | <b>Próxima Calibração:</b> {str(row['DATA_PROXIMA'])[:10]}</p>
        </div>
        """, unsafe_allow_html=True)
